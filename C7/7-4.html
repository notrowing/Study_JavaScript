● JavaScript Old API (POST)

<script>
  const xhr = new XMLHttpRequest();
  xhr.onreadystatechange = callback;
  xhr.open('post', '주소');

  xhr.setRequestHeader('key', 'value');

  const formData = new FormData();
  formData.append('key', 'value');
  xhr.send(formData);
  function callback() {
    if (xhr.readyState == 4) {
      if (xhr.status == 200) {
        let result = xhr.response;
        console.log(result);
      }
    }
  }
</script>
